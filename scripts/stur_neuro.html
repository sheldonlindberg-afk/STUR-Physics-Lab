#!/usr/bin/env python3
# ===============================================================
# STURâ€“Neuroplasticity Simulation â€“ Unified Resistance Lagrangian
# ===============================================================
# JupyterLite-safe numerical integration of the radial motion
# under curvature-regulated STUR potential + spin-field forcing.
# ===============================================================

import numpy as np, math, matplotlib.pyplot as plt

# --- Physical parameters ----------------------------------------
mu = 1.0
ke, G = 8.9875e9, 6.674e-11
q1, q2 = 1.0e-6, -1.0e-6
m1, m2 = 1.0, 1.0
alpha_s, sigma = 0.2, 0.05
L0, lam, ellB = 1.0, 2.0, 1.5
gamma0, B0, omega = 0.1, 1.0, 2.0*math.pi*0.2

# --- STUR modulator and potential --------------------------------
def S_STUR(r,L0): return math.tanh(r/L0)*(1-math.exp(-r/L0))
def dU_dr(r):
    if r==0: return 0
    S=S_STUR(r,L0)
    term=(ke*q1*q2 - G*m1*m2)/r**2 + alpha_s*((1-math.exp(-r/lam))/r**2 + sigma*r)
    return -S*term
def U_STUR(r):
    # numerical integration for plotting
    rs=np.linspace(1e-3,r,500)
    integrand=[S_STUR(rp,L0)*((ke*q1*q2 - G*m1*m2)/rp**2 +
               alpha_s*((1-math.exp(-rp/lam))/rp**2 + sigma*rp)) for rp in rs]
    return -np.trapz(integrand,rs)

# --- Time integration --------------------------------------------
Tmax, dt = 200.0, 0.01
N = int(Tmax/dt)
t = np.linspace(0,Tmax,N)
r, v = np.zeros(N), np.zeros(N)
r[0]=1.0; v[0]=0.0

for i in range(N-1):
    F = -dU_dr(r[i]) + gamma0*math.exp(-r[i]/ellB)*B0*math.sin(omega*t[i])
    a = F/mu
    v[i+1] = v[i] + a*dt
    r[i+1] = max(r[i] + v[i+1]*dt, 1e-6)

# --- Plot results -------------------------------------------------
Uvals = [U_STUR(rr) for rr in np.linspace(0.1,3.0,300)]

plt.figure(figsize=(10,8))
plt.subplot(2,1,1)
plt.plot(np.linspace(0.1,3.0,300),Uvals,color='cyan')
plt.title('STURâ€“Neuroplastic Potential U(r)')
plt.xlabel('r'); plt.ylabel('U(r)')

plt.subplot(2,1,2)
plt.plot(t,r,color='#ffaa00',label='r(t)')
plt.title('Radial Response under Spinâ€“Field Forcing')
plt.xlabel('Time'); plt.ylabel('r(t)')
plt.grid(True,alpha=0.3); plt.tight_layout()
plt.show()
