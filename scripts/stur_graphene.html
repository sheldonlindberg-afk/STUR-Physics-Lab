import numpy as np, math, json, matplotlib.pyplot as plt

# ----------------- Fundamental constants ------------------------
hbar   = 1.055e-34      # J*s
c      = 3.0e8          # m/s
e_ch   = 1.602e-19      # C
kB     = 1.38e-23       # J/K
pi     = math.pi

# ----------------- Fuzzy DM force model -------------------------
# Simple Yukawa-like DM force:
#   F_DM(r) = lambda_DM * phi^2 * exp(-r / ell_DM) / r^2
# Units are normalized; we care about relative suppression.
lambda_DM = 1.0    # dimensionless coupling
phi0      = 1.0    # background scalar amplitude
ell_DM    = 1.0e20 # effective range (m), ~ kpc-scale for illustration

def F_DM(r):
    r_arr = np.asarray(r, dtype=float)
    r_eff = np.maximum(r_arr, 1e-6)
    return lambda_DM * (phi0**2) * np.exp(-r_eff/ell_DM) / (r_eff**2)

# Radii from 0.1 kpc to 2 Mpc
kpc = 3.086e19
r_DM = np.logspace(-1, 3.3, 200) * kpc  # 0.1 kpc to ~2e3 kpc
F_DM_vals = F_DM(r_DM)

# Suppression ratio at 10 kpc vs 1 kpc
r_1kpc  = 1.0 * kpc
r_10kpc = 10.0 * kpc
F_1kpc  = float(F_DM(r_1kpc))
F_10kpc = float(F_DM(r_10kpc))
supp_ratio_10kpc = F_10kpc / F_1kpc if F_1kpc != 0 else 0.0

# ----------------- Graphene transport model ---------------------
T = 10.0                     # K
n_cm2 = 1.8e12               # cm^-2
n = n_cm2 * 1e4              # m^-2
vF = 1.0e6                   # m/s (Dirac v_F)
g_s, g_v = 2.0, 2.0          # spin, valley

# Fermi wavevector and energy
kF = math.sqrt(pi * n)       # [m^-1], using n = kF^2 / Ï€ for graphene
E_F = hbar * vF * kF         # J

# Effective mass for Drude mapping: m_eff ~ E_F / vF^2
m_eff = E_F / (vF**2)

# Baseline scattering time (order-of-magnitude)
tau0 = 1.0e-13               # s, ~0.1 ps scale at low T
B = 10.0                     # Tesla

# Cyclotron frequency and baseline conductivities
def sigma_xx_from_tau(tau):
    return n * e_ch**2 * tau / m_eff

def sigma_xy_from_tau(tau):
    omega_c = e_ch * B / m_eff
    denom = 1.0 + (omega_c * tau)**2
    return n * e_ch**2 * tau * omega_c / (m_eff * denom)

sigma_xx0 = sigma_xx_from_tau(tau0)
sigma_xy0 = sigma_xy_from_tau(tau0)

# ----------------- Axion coupling -> Î”Ï„/Ï„ -----------------------
# Phenomenological scaling:
#   Î”Ï„/Ï„ â‰ˆ C * (g_a / g0)^p
# with g0 = 1e-12, C = 5e-6, p â‰ˆ 0.65 chosen so that:
#   g_a = 1e-12 -> Î”Ï„/Ï„ â‰ˆ 5e-6  (~0.0005%)
#   g_a = 1e-10 -> Î”Ï„/Ï„ â‰ˆ 1e-4 (~0.01%)
g0_ref = 1.0e-12
C_ref  = 5.0e-6
p_ref  = 0.65

def delta_tau_frac(g_a):
    g_a = float(g_a)
    return C_ref * (g_a / g0_ref)**p_ref

def tau_from_g(g_a):
    return tau0 * (1.0 + delta_tau_frac(g_a))

def sigma_xx(g_a):
    return sigma_xx_from_tau(tau_from_g(g_a))

def sigma_xy(g_a):
    return sigma_xy_from_tau(tau_from_g(g_a))

# g_a sweep
g_vals = np.logspace(-13, -9, 120)
d_tau_frac = np.array([delta_tau_frac(g) for g in g_vals])
sigma_xx_vals = np.array([sigma_xx(g) for g in g_vals])
sigma_xy_vals = np.array([sigma_xy(g) for g in g_vals])

d_sigma_xx_frac = (sigma_xx_vals - sigma_xx0) / sigma_xx0
d_sigma_xy_frac = (sigma_xy_vals - sigma_xy0) / sigma_xy0

# Special points: g_a = 1e-12, 1e-10
g1, g2 = 1.0e-12, 1.0e-10

def eval_point(g):
    dt  = delta_tau_frac(g)
    sxx = sigma_xx(g)
    sxy = sigma_xy(g)
    return {
        "g_a": g,
        "delta_tau_over_tau": float(dt),
        "delta_sigma_xx_over_sigma_xx": float((sxx - sigma_xx0)/sigma_xx0),
        "delta_sigma_xy_over_sigma_xy": float((sxy - sigma_xy0)/sigma_xy0)
    }

pt1 = eval_point(g1)
pt2 = eval_point(g2)

# ----------------- Plotting -------------------------------------
# (1) Fuzzy DM force profile
plt.figure(figsize=(9,4))
plt.loglog(r_DM/kpc, np.abs(F_DM_vals), color="cyan", label="|F_DM(r)|")
plt.axvline(1.0, color="gray", ls="--", alpha=0.5)
plt.axvline(10.0, color="gray", ls="--", alpha=0.5)
plt.text(1.1, np.max(np.abs(F_DM_vals))*0.5, "1 kpc", color="gray")
plt.text(10.1, np.max(np.abs(F_DM_vals))*0.3, "10 kpc", color="gray")
plt.xlabel("r (kpc)")
plt.ylabel("|F_DM(r)| (arb. units)")
plt.title("Fuzzy DM Yukawa Force Profile")
plt.grid(True, which="both", ls=":")
plt.tight_layout()
plt.show()

# (2) Graphene conductivity shifts vs g_a
plt.figure(figsize=(9,5))
plt.loglog(g_vals, np.abs(d_tau_frac), label="|Î”Ï„/Ï„|")
plt.loglog(g_vals, np.abs(d_sigma_xx_frac), label="|Î”Ïƒ_xx/Ïƒ_xx|", ls="--")
plt.loglog(g_vals, np.abs(d_sigma_xy_frac), label="|Î”Ïƒ_xy/Ïƒ_xy|", ls=":")
plt.axvline(g1, color="gray", ls="--", alpha=0.5)
plt.axvline(g2, color="gray", ls="--", alpha=0.5)
plt.text(g1*1.05, 1e-7, "g_a=1e-12", color="gray")
plt.text(g2*1.05, 1e-7, "g_a=1e-10", color="gray")
plt.xlabel("axion coupling g_a")
plt.ylabel("fractional change (absolute value)")
plt.title("Graphene response vs axion coupling g_a")
plt.grid(True, which="both", ls=":")
plt.legend()
plt.tight_layout()
plt.show()

# ----------------- JSON summary ---------------------------------

    "fuzzy_DM": {
        "ell_DM_m": float(ell_DM),
        "F_DM_1kpc": F_1kpc,
        "F_DM_10kpc": F_10kpc,
        "suppression_10kpc_over_1kpc": supp_ratio_10kpc
    },
    "graphene": {
        "T_K": T,
        "n_cm2": n_cm2,
        "E_F_J": float(E_F),

        },
        "points": {
            "g_a_1e-12": pt1,
            "g_a_1e-10": pt2
        }
    }
}
