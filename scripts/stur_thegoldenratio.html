# STUR Perfect Resistance Ratio Simulation ‚Äì R* = E / (I Œ¶)

import numpy as np
import matplotlib.pyplot as plt
import json

plt.style.use("dark_background")
np.random.seed(42)

# ---------------------- PARAMETERS ------------------------------
# Base energy, information, and flux scales
E0   = 1.0          # normalized energy unit
I0   = 1.0          # normalized information flow
Phi0 = 1.0          # normalized geometric flux

# Oscillation & coupling constants
omega_E   = 2.0 * np.pi * 0.25   # Hz equivalent
omega_I   = 2.0 * np.pi * 0.27
omega_Phi = 2.0 * np.pi * 0.23
k_couple  = 0.15                 # coupling factor for feedback
noise     = 0.02

# Time domain
t = np.linspace(0, 60, 2000)  # 60 s, fine resolution

# ---------------------- DYNAMICS -------------------------------
# Coherent oscillations with slight phase drifts
E   = E0   * (1 + 0.3*np.sin(omega_E*t))
I   = I0   * (1 + 0.3*np.sin(omega_I*t + np.pi/3))
Phi = Phi0 * (1 + 0.3*np.sin(omega_Phi*t + np.pi/5))

# Feedback coupling: small self-correction toward unity
for _ in range(3):
    R = E / (I*Phi)
    E -= k_couple * (R - 1) * E + noise*np.random.randn(len(E))*1e-2

# Recompute final ratio
R_star = E / (I * Phi)

# ---------------------- OUTPUTS -------------------------------
R_mean = np.mean(R_star)
R_std  = np.std(R_star)

summary = {
    "R_mean": float(R_mean),
    "R_std": float(R_std),
    "E_mean": float(np.mean(E)),
    "I_mean": float(np.mean(I)),
    "Phi_mean": float(np.mean(Phi)),
    "Interpretation": "R* ‚âà 1 indicates perfect coherence between energy, information, and geometry."
}

print(json.dumps(summary, indent=2))

# ---------------------- PLOT ----------------------------------
fig, ax = plt.subplots(figsize=(8,4))
ax.plot(t, R_star, color="#ffaa00", lw=1.5, label=r"$R_\ast = E/(I\Phi)$")
ax.axhline(1, color="cyan", ls="--", lw=1, label="Perfect Coherence (R*=1)")
ax.set_xlabel("Time (s)")
ax.set_ylabel("R* ratio")
ax.set_title("STUR Perfect Resistance Ratio Simulation")
ax.legend()
plt.show()
